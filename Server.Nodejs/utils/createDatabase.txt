CREATE TABLE shed(shedID INTEGER PRIMARY KEY AUTOINCREMENT, name VARCHAR(50), description VARCHAR(50));

CREATE TABLE room(roomID INTEGER PRIMARY KEY AUTOINCREMENT, name VARCHAR(50), description VARCHAR(50), shedID INTEGER, FOREIGN KEY(shedID) REFERENCES shed(shedID));

CREATE TABLE sensor(sensorID INTEGER PRIMARY KEY AUTOINCREMENT, code VARCHAR(50) NOT NULL, description VARCHAR(50), roomID INTEGER NOT NULL, FOREIGN KEY(roomID) REFERENCES room(roomID));

CREATE TABLE dataType(dataTypeID INTEGER PRIMARY KEY AUTOINCREMENT, name VARCHAR(50) NOT NULL);

CREATE TABLE measure(measureID INTEGER PRIMARY KEY AUTOINCREMENT, data REAL NOT NULL, timeLog DATETIME NOT NULL, sensorID INTEGER NOT NULL, dataTypeID INTEGER NOT NULL, FOREIGN KEY(sensorID) REFERENCES sensor(sensorID), FOREIGN KEY(dataTypeID) REFERENCES dataType(dataTypeID));

CREATE TABLE nonRegisteredSensor(nonRegisteredSensorID INTEGER PRIMARY KEY AUTOINCREMENT, code VARCHAR(50) NOT NULL, timeReport DATETIME NOT NULL);
